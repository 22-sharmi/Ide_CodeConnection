Arduino UNO - Reaction Time Game (Joystick)

Test your reflexes using a joystick connected to Arduino UNO!

Reaction Time Game – Detailed Notes

1. Project Overview

This project measures how quickly you can react after a random delay. When the Serial Monitor displays “GO!”, move the joystick as fast as possible. Your reaction time (in milliseconds) will be shown on the Serial Monitor.

2. How It Works

  • The player presses the joystick button to start.
  • Arduino waits for a random time (between 1–5 seconds).
  • Then it displays “GO!” — and starts a timer using millis().
  • As soon as the joystick is moved, Arduino calculates the time difference to determine the player’s reaction speed.

3. Circuit Connections

  • VRx → A0
  • VRy → A1
  • SW (Button) → D2
  • VCC → 5V
  • GND → GND

4. Key Functions Used

  • analogRead() – Reads the X and Y axis values from the joystick.
  • digitalRead() – Detects the joystick button press.
  • millis() – Tracks elapsed time in milliseconds (used for timing reaction).
  • random() – Generates a random delay between 1 and 5 seconds.

5. Applications

  • Reflex testing games
  • Human reaction experiments
  • Fun challenge projects with friends

// Reaction Time Game Using Joystick

const int VRx = A0;
const int VRy = A1;
const int SW = 2;

int xCenter, yCenter;
bool waitingToStart = true;

void setup() {
  Serial.begin(9600);
  pinMode(SW, INPUT_PULLUP);
  Serial.println("=== Reaction Time Game ===");
  Serial.println("Press the joystick button to start...");

  // Read center positions for calibration
  xCenter = analogRead(VRx);
  yCenter = analogRead(VRy);
}

void loop() {
  // Wait for the player to press the joystick button
  if (waitingToStart && digitalRead(SW) == LOW) {
    waitingToStart = false;
    Serial.println("\nGet ready...");
    delay(1000);

    // Wait for a random time (1–5 seconds)
    // unsigned long is a variable type that can hold large positive integers.
    unsigned long randomDelayTime = random(1000, 5000); // Pick a random wait time (1–5 seconds)
    delay(randomDelayTime);

    Serial.println("GO! Move the joystick!");

    unsigned long startTime = millis(); // Returns the number of milliseconds since the program started (like a stopwatch)

    // Wait until the player moves the joystick
    while (true) {
      int xVal = analogRead(VRx);
      int yVal = analogRead(VRy);

      if (abs(xVal - xCenter) > 150 || abs(yVal - yCenter) > 150) {  // Detect joystick movement (any direction)
        unsigned long reactionTime = millis() - startTime;
        Serial.print("Your reaction time: ");
        Serial.print(reactionTime);
        Serial.println(" ms");
        Serial.println("Press the joystick button to play again...");
        waitingToStart = true;
        delay(1000);
        break;
      }
    }
  }
}
