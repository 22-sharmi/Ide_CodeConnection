Smart Climate Notifier (UNO)

DHT11 sensor checks temperature and humidity. LED & buzzer turn ON when temp > 30°C or humidity > 70%.

Circuit Connection:

  • DHT11 Data pin → Digital pin 2
  • LED → pin 8
  • Buzzer → pin 7
  • VCC → 5V, GND → GND

#include <DHT.h>
#define DHTPIN 2
#define DHTTYPE DHT11
DHT dht(DHTPIN, DHTTYPE);
int led = 8;
int buzzer = 7;

void setup() {
  Serial.begin(9600);
  dht.begin();
  pinMode(led, OUTPUT);
  pinMode(buzzer, OUTPUT);
}

void loop() {
  float temp = dht.readTemperature();
  float hum = dht.readHumidity();
  Serial.print("Temp: ");
  Serial.print(temp);
  Serial.print(" °C, Humidity: ");
  Serial.print(hum);
  Serial.println(" %");

  if (temp > 30 || hum > 70) {
    digitalWrite(led, HIGH);
    digitalWrite(buzzer, HIGH);
  } else {
    digitalWrite(led, LOW);
    digitalWrite(buzzer, LOW);
  }
  delay(2000);
}
