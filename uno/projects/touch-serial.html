Arduino UNO - Touch Sensor Serial

Displays “Touched” or “Not Touched” on Serial Monitor based on sensor input.


int touchPin = 2; // Connect touch sensor output to pin 2

void setup() {
  pinMode(touchPin, INPUT);
  Serial.begin(9600);
}

void loop() {
  int touchValue = digitalRead(touchPin);
  if (touchValue == HIGH) {
    Serial.println("Touched");
  } else {
    Serial.println("Not Touched");
  }
  delay(500);
}
