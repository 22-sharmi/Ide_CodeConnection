Arduino UNO - Serial Toggle

Control LED and buzzer using Serial Monitor. Send '1' to turn ON, '0' to turn OFF.


int ledPin = 13;
int buzzer = 8;

void setup() {
  pinMode(ledPin, OUTPUT);
  pinMode(buzzer, OUTPUT);
  Serial.begin(9600);
}

void loop() {
  if (Serial.available()) {
    char cmd = Serial.read();
    if (cmd == '1') {
      digitalWrite(ledPin, HIGH);
      digitalWrite(buzzer, HIGH);
    } else if (cmd == '0') {
      digitalWrite(ledPin, LOW);
      digitalWrite(buzzer, LOW);
    }
  }
}
