Arduino UNO - Vibration Sensor

Detect vibrations using the SW-420 sensor and print to Serial Monitor.

Circuit Connection:

  • Sensor OUT to digital pin 6
  • VCC to 5V
  • GND to GND

int vibrationPin = 6;

void setup() {
  pinMode(vibrationPin, INPUT);
  Serial.begin(9600);
}

void loop() {
  int vib = digitalRead(vibrationPin);
  if (vib == HIGH) {
    Serial.println("Vibration Detected!");
  } else {
    Serial.println("No Vibration");
  }
  delay(500);
}
