Auto Light Motion Detector

PIR sensor turns on LED light when motion is detected.

Circuit Connection:

  • PIR OUT to digital pin 8
  • LED +ve to digital pin 7 via 220Ω resistor
  • PIR VCC to 5V, GND to GND

int pirPin = 8;
int ledPin = 7;

void setup() {
  pinMode(pirPin, INPUT);
  pinMode(ledPin, OUTPUT);
  Serial.begin(9600);
}

void loop() {
  int motion = digitalRead(pirPin);
  if (motion == HIGH) {
    digitalWrite(ledPin, HIGH);
    Serial.println("Motion Detected - Light ON");
  } else {
    digitalWrite(ledPin, LOW);
    Serial.println("No Motion - Light OFF");
  }
  delay(500);
}
