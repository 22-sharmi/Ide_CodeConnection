Earthquake Vibration Alert

Triggers buzzer and LED on vibration detection.

Circuit Connection:

  • Sensor OUT to pin 4
  • LED to pin 3 via 220Ω resistor
  • Buzzer to pin 2
  • VCC to 5V, GND to GND

int vibPin = 4;
int ledPin = 3;
int buzzer = 2;

void setup() {
  pinMode(vibPin, INPUT);
  pinMode(ledPin, OUTPUT);
  pinMode(buzzer, OUTPUT);
  Serial.begin(9600);
}

void loop() {
  int vib = digitalRead(vibPin);
  if (vib == HIGH) {
    digitalWrite(ledPin, HIGH);
    digitalWrite(buzzer, HIGH);
    Serial.println("Vibration Detected!");
  } else {
    digitalWrite(ledPin, LOW);
    digitalWrite(buzzer, LOW);
    Serial.println("No Vibration");
  }
  delay(500);
}
