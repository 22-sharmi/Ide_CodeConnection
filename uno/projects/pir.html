Arduino UNO - PIR Motion Sensor

Detect motion and print the result to Serial Monitor.

Circuit Connection:

  • PIR OUT to digital pin 8
  • VCC to 5V
  • GND to GND

int pirPin = 8;

void setup() {
  pinMode(pirPin, INPUT);
  Serial.begin(9600);
}

void loop() {
  int motion = digitalRead(pirPin);
  if (motion == HIGH) {
    Serial.println("Motion Detected!");
  } else {
    Serial.println("No Motion");
  }
  delay(1000);
}
