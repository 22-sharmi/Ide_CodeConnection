Arduino UNO - IR Obstacle Sensor

Detect obstacles using IR sensor and print to Serial Monitor.

Circuit Connection:

  • IR OUT to digital pin 7
  • VCC to 5V
  • GND to GND

int irPin = 7;

void setup() {
  pinMode(irPin, INPUT);
  Serial.begin(9600);
}

void loop() {
  int obstacle = digitalRead(irPin);
  if (obstacle == LOW) {
    Serial.println("Obstacle Detected!");
  } else {
    Serial.println("No Obstacle");
  }
  delay(500);
}
