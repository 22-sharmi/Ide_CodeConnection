Arduino UNO - Serial Control Game

Guess the Number Game Using Serial Monitor
LED ON For Correct Answer , Buzzer For Wrong !


const int ledPin = 13;
const int buzzerPin = 12;
int secretNumber;

void setup() {
  pinMode(ledPin, OUTPUT);
  pinMode(buzzerPin, OUTPUT);
  Serial.begin(9600);
  randomSeed(analogRead(0)); // Seed random with analog noise
  secretNumber = random(1, 6); // 1 to 5
  Serial.println("ğŸ® Guess the secret number between 1 and 5!");
}

void loop() {
  if (Serial.available()) {
    int guess = Serial.parseInt();
    
    if (guess == secretNumber) {
      Serial.println("ğŸ‰ Correct! You win!");
      for (int i = 0; i < 5; i++) {
        digitalWrite(ledPin, HIGH);
        delay(100);
        digitalWrite(ledPin, LOW);
        delay(100);
      }
      secretNumber = random(1, 6); // Pick new number
      Serial.println("ğŸ” New number generated! Guess again!");
    } else {
      Serial.println("âŒ Wrong! Try again!");
      digitalWrite(buzzerPin, HIGH);
      delay(300);
      digitalWrite(buzzerPin, LOW);
    }
  }
}
