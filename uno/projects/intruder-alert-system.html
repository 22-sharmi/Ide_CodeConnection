Intruder Alert System

IR sensor activates buzzer when obstacle is detected.

Circuit Connection:

  • IR OUT to digital pin 6
  • Buzzer +ve to digital pin 5, -ve to GND
  • IR VCC to 5V, GND to GND

int irPin = 6;
int buzzer = 5;

void setup() {
  pinMode(irPin, INPUT);
  pinMode(buzzer, OUTPUT);
  Serial.begin(9600);
}

void loop() {
  int obstacle = digitalRead(irPin);
  if (obstacle == LOW) {
    digitalWrite(buzzer, HIGH);
    Serial.println("Intruder Detected!");
  } else {
    digitalWrite(buzzer, LOW);
    Serial.println("No Intruder");
  }
  delay(500);
}
