ESP32 - Touch Sensor

Monitor touch sensor readings via Serial Monitor. The message "Touched" will appear when touched, otherwise "Not Touched".

Circuit Connection:

  • Touch Sensor OUT pin to GPIO 2
  • VCC to 3.3V
  • GND to GND

int touchPin = 2; // Connect touch sensor output to pin 2

void setup() {
  pinMode(touchPin, INPUT);
  Serial.begin(9600);
}

void loop() {
  int touchValue = digitalRead(touchPin);
  if (touchValue == HIGH) {
    Serial.println("Touched");
  } else {
    Serial.println("Not Touched");
  }
  delay(500);
}
