ESP32 - Ultrasonic Sensor

Measure distance using HC-SR04 and view in Serial Monitor.

Circuit Connection:

  • Trig to GPIO 5
  • Echo to GPIO 18
  • VCC to 5V
  • GND to GND

#define trigPin 5
#define echoPin 18

long duration;
int distance;

void setup() {
  pinMode(trigPin, OUTPUT);
  pinMode(echoPin, INPUT);
  Serial.begin(115200);
}

void loop() {
  digitalWrite(trigPin, LOW);
  delayMicroseconds(2);
  digitalWrite(trigPin, HIGH);
  delayMicroseconds(10);
  digitalWrite(trigPin, LOW);

  duration = pulseIn(echoPin, HIGH);
  distance = duration * 0.034 / 2;

  Serial.print("Distance: ");
  Serial.print(distance);
  Serial.println(" cm");

  delay(1000);
}
