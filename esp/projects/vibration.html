ESP32 - Vibration Sensor

Monitor physical vibration using Serial Monitor.

Circuit Connection:

  • Vibration OUT to GPIO 14
  • VCC to 3.3V
  • GND to GND

int vibrationPin = 14;

void setup() {
  pinMode(vibrationPin, INPUT);
  Serial.begin(115200);
}

void loop() {
  int vib = digitalRead(vibrationPin);
  if (vib == HIGH) {
    Serial.println("Vibration Detected!");
  } else {
    Serial.println("No Vibration");
  }
  delay(500);
}
