ESP32 - IR Obstacle Sensor

Detect obstacles using IR sensor and Serial Monitor.

Circuit Connection:

  • IR OUT to GPIO 12
  • VCC to 3.3V or 5V
  • GND to GND

int irPin = 12;

void setup() {
  pinMode(irPin, INPUT);
  Serial.begin(115200);
}

void loop() {
  int obstacle = digitalRead(irPin);
  if (obstacle == LOW) {
    Serial.println("Obstacle Detected!");
  } else {
    Serial.println("No Obstacle");
  }
  delay(500);
}
