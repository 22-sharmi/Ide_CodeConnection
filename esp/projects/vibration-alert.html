Earthquake Vibration Alert (ESP32)

Triggers buzzer and LED on vibration detection using ESP32.

Circuit Connection (ESP32):

  • Vibration Sensor OUT → GPIO 4
  • LED → GPIO 3 via 220Ω resistor
  • Buzzer → GPIO 2
  • VCC → 3.3V, GND → GND

// Earthquake Vibration Alert using ESP32

const int vibPin = 4;    // Vibration sensor
const int ledPin = 3;    // LED
const int buzzer = 2;    // Buzzer

void setup() {
  Serial.begin(115200);
  pinMode(vibPin, INPUT);
  pinMode(ledPin, OUTPUT);
  pinMode(buzzer, OUTPUT);
}

void loop() {
  int vib = digitalRead(vibPin);
  if (vib == HIGH) {
    digitalWrite(ledPin, HIGH);
    digitalWrite(buzzer, HIGH);
    Serial.println("⚠️ Vibration Detected!");
  } else {
    digitalWrite(ledPin, LOW);
    digitalWrite(buzzer, LOW);
    Serial.println("No Vibration");
  }
  delay(500);
}
