Smart Distance Alarm - ESP32

Ultrasonic sensor triggers alert when object is too close (ESP32 Version).

Circuit Connection:

  • Trig to GPIO 12, Echo to GPIO 14
  • LED to GPIO 26, Buzzer to GPIO 27
  • VCC to 3.3V or 5V, GND to GND

#define trigPin 12
#define echoPin 14
#define led 26
#define buzzer 27

void setup() {
  pinMode(trigPin, OUTPUT);
  pinMode(echoPin, INPUT);
  pinMode(led, OUTPUT);
  pinMode(buzzer, OUTPUT);
  Serial.begin(115200);
}

void loop() {
  digitalWrite(trigPin, LOW);
  delayMicroseconds(2);
  digitalWrite(trigPin, HIGH);
  delayMicroseconds(10);
  digitalWrite(trigPin, LOW);

  long duration = pulseIn(echoPin, HIGH);
  int distance = duration * 0.034 / 2;

  Serial.print("Distance: ");
  Serial.print(distance);
  Serial.println(" cm");

  if (distance < 15) {
    digitalWrite(led, HIGH);
    digitalWrite(buzzer, HIGH);
  } else {
    digitalWrite(led, LOW);
    digitalWrite(buzzer, LOW);
  }

  delay(500);
}
