ESP32 - Reaction Time Game (Push Button)

Measure how fast you react when the LED lights up using an ESP32 and push button!

Reaction Time Game – ESP32 Notes

1. Project Overview

This ESP32 project measures your reaction speed by timing how quickly you press a button after an LED lights up.

2. How It Works

  • The ESP32 waits for a random delay between 2–5 seconds.
  • When the LED turns on, a timer starts using millis().
  • When you press the button, ESP32 calculates your reaction time.

3. ESP32 Circuit Connections

  • LED → GPIO 2 (built-in LED on many ESP32 boards)
  • Push Button → GPIO 4
  • Other button pin → GND
  • No resistor needed (ESP32 uses internal pull-up)

4. Key ESP32 Notes

  • ESP32 uses esp_random() to seed randomness.
  • Input mode INPUT_PULLUP works the same as Arduino.
  • Serial speed is faster (115200 baud).

5. Applications

  • Reflex testing
  • Simple reaction games
  • Learning ESP32 input/output fundamentals

// ESP32 Reaction Time Game Using Push Button and LED

const int ledPin = 2;       // LED on GPIO2 (often built-in)
const int buttonPin = 4;    // Push button on GPIO4

unsigned long startTime;
unsigned long reactionTime;
bool waitingForPress = false;

void setup() {
  pinMode(ledPin, OUTPUT);
  pinMode(buttonPin, INPUT_PULLUP);  // Button with internal pull-up

  Serial.begin(115200);
  delay(1000);

  // ESP32 hardware random seed
  randomSeed(esp_random());

  Serial.println("=== ESP32 Reaction Time Game ===");
  Serial.println("Wait for the LED, then press the button!");
}

void loop() {
  delay(random(2000, 5000)); // Random wait between 2–5 seconds

  digitalWrite(ledPin, HIGH);   // LED ON
  startTime = millis();         // Start timer
  waitingForPress = true;

  while (waitingForPress) {
    if (digitalRead(buttonPin) == LOW) { // Button pressed
      reactionTime = millis() - startTime;

      Serial.print("Your reaction time: ");
      Serial.print(reactionTime);
      Serial.println(" ms");

      waitingForPress = false;
      digitalWrite(ledPin, LOW);  // LED OFF
      delay(2000);                // Pause before next round
    }
  }
}
