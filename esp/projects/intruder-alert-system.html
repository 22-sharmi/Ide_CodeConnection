Intruder Alert System (ESP32)

IR sensor activates buzzer when obstacle is detected using ESP32.

Circuit Connection:

  • IR OUT to GPIO 18
  • Buzzer +ve to GPIO 19, -ve to GND
  • IR VCC to 3.3V (or 5V if supported), GND to GND

#define irPin 18
#define buzzer 19

void setup() {
  pinMode(irPin, INPUT);
  pinMode(buzzer, OUTPUT);
  Serial.begin(115200);
}

void loop() {
  int obstacle = digitalRead(irPin);
  if (obstacle == LOW) {
    digitalWrite(buzzer, HIGH);
    Serial.println("Intruder Detected!");
  } else {
    digitalWrite(buzzer, LOW);
    Serial.println("No Intruder");
  }
  delay(500);
}
