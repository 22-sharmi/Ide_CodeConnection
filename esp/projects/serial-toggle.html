ESP32 - Serial Monitor

Control LED and Buzzer using Serial Monitor. Send '1' to turn ON, '0' to turn OFF.

Circuit Connection:

  • LED +ve to GPIO 2
  • LED -ve to 220Ω resistor to GND
  • Buzzer +ve to GPIO 8
  • Buzzer -ve to GND

int ledPin = 2;
int buzzer = 8;

void setup() {
  pinMode(ledPin, OUTPUT);
  pinMode(buzzer, OUTPUT);
  Serial.begin(115200);
}

void loop() {
  if (Serial.available()) {
    char cmd = Serial.read();
    if (cmd == '1') {
      digitalWrite(ledPin, HIGH);
      digitalWrite(buzzer, HIGH);
    } else if (cmd == '0') {
      digitalWrite(ledPin, LOW);
      digitalWrite(buzzer, LOW);
    }
  }
}
